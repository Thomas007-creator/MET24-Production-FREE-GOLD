            </Card>
          ))}
        </div>
 
        {/* Action Buttons */}
        <div className='flex justify-center gap-4'>

            startContent={<Download className='w-4 h-4' />}
            className='bg-purple-600 hover:bg-purple-700'
          >

          <Button

            Deel Resultaten
          </Button>
          <Button
            color='secondary'

          >
            Update Scores
          </Button>
        </div>
      </div>
    </div>
  );
